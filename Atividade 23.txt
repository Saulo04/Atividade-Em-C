#include <stdio.h>

int main() {
    int valor, maior, menor;

    printf("Digite o 1ยบ valor: ");
    scanf("%d", &valor);

    maior = menor = valor;  // Inicializa com o primeiro valor

    for (int i = 2; i <= 50; i++) {
        printf("Digite o %dยบ valor: ", i);
        scanf("%d", &valor);

        if (valor > maior) {
            maior = valor;
        }

        if (valor < menor) {
            menor = valor;
        }
    }

    printf("\nMaior valor informado: %d\n", maior);
    printf("Menor valor informado: %d\n", menor);

    return 0;
}

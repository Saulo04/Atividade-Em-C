#include <stdio.h>

int main() {
    int num, i;

    // Entrada do n√∫mero
    printf("Digite um numero para ver a tabuada: ");
    scanf("%d", &num);

    // Loop para calcular e mostrar a tabuada de 1 a 10
    for(i = 1; i <= 10; i++) {
        printf("%d x %d = %d\n", num, i, num * i);
    }

    return 0;
}
